import { Resend } from 'resend';

const resend = new Resend(process.env.RESEND_API_KEY);

    const { email, source, data } = await request.json();

    // Store in Supabase

        firstName: data.businessName,

    return Response.json({ success: true });
}

## Email Best Practices

1. **Send from real name** (not "noreply")
2. **Keep under 200 words**
3. **One clear CTA per email**
4. **Mobile-first formatting**
5. **Plain text version always**
6. **Unsubscribe link required**
7. **Test subject lines** (A/B)
8. **Send at 10am local time**
